// Using remove() method
function removeElementWithRemove() {
    const element = document.getElementById('element-to-remove');
    element.remove(); // Directly removes the element
}

// Using removeChild() method
function removeElementWithRemoveChild() {
    const parent = document.getElementById('parent-element');
    const child = document.getElementById('child-element');
    parent.removeChild(child); // Removes the child from parent
}

// Remove with animation
function removeWithAnimation(element) {
    element.style.animation = 'fadeOut 0.5s ease forwards';
    setTimeout(() => {
        element.remove();
    }, 500);
}

// Check if element exists before removing
function safeRemove(element) {
    if (element && element.parentNode) {
        element.parentNode.removeChild(element);
    }
}

DOM Removal Methods

JavaScript provides two primary methods for removing elements from the DOM:

  • remove(): Removes the element directly from the DOM. This is a newer method that doesn't require reference to the parent element.
  • removeChild(): Removes a child element from the DOM. Requires reference to both the parent element and the child element to be removed.

Key differences:

remove() is more straightforward and doesn't require access to the parent element. removeChild() is older but still widely used, and it returns a reference to the removed node.

JavaScript Code: Practical Examples 
// Create multiple elements
function createMultipleElements(count) {
    const container = document.getElementById('dynamicContainer');
    container.innerHTML = ''; // Clear existing elements
    
    for (let i = 0; i < count; i++) {
        const element = document.createElement('div');
        element.className = 'element-item';
        element.textContent = `Element ${i+1}`;
        element.id = `dynamic-${i+1}`;
        container.appendChild(element);
    }
}

// Remove the last element
function removeLastElement() {
    const container = document.getElementById('dynamicContainer');
    if (container.lastChild) {
        container.lastChild.remove();
    }
}

// Remove a random element
function removeRandomElement() {
    const container = document.getElementById('dynamicContainer');
    const children = container.children;
    
    if (children.length > 0) {
        const randomIndex = Math.floor(Math.random() * children.length);
        children[randomIndex].remove();
    }
}
