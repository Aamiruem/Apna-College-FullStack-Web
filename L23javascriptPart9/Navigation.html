JavaScript Code Example

// Changing background color
const bgColorPicker = document.getElementById('bgColor');
const targetElement = document.getElementById('styleTarget');

bgColorPicker.addEventListener('input', () => {
    targetElement.style.backgroundColor = bgColorPicker.value;
});

// Adjusting width
const widthSlider = document.getElementById('widthRange');
const widthValue = document.getElementById('widthValue');
const sizeTarget = document.getElementById('sizeTarget');

widthSlider.addEventListener('input', () => {
    sizeTarget.style.width = `${widthSlider.value}%`;
    widthValue.textContent = `${widthSlider.value}%`;
});

// Changing font family
const fontSelector = document.getElementById('fontFamily');
const textTarget = document.getElementById('textTarget');

fontSelector.addEventListener('change', () => {
    textTarget.style.fontFamily = fontSelector.value;
});
