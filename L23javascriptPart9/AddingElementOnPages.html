// Get the current element
const currentElement = document.getElementById('current');

// Navigate to parent
const parent = currentElement.parentElement;

// Navigate to children
const children = currentElement.children;

// Navigate to previous sibling
const prevSibling = currentElement.previousElementSibling;

// Navigate to next sibling
const nextSibling = currentElement.nextElementSibling;

// Example: Change background of parent
if (parent) {
    parent.style.backgroundColor = '#fdbb2d';
}

// Example: Loop through children
for (let i = 0; i < children.length; i++) {
    children[i].style.fontWeight = 'bold';
}

DOM Navigation Properties

JavaScript provides several properties to navigate between DOM elements:

  • parentElement: Returns the parent element of the specified element
  • children: Returns a collection of an element's child elements
  • previousElementSibling: Returns the previous element in the same tree level
  • nextElementSibling: Returns the next element in the same tree level

These properties are read-only and return null if no such element exists.

JavaScript Code: Practical Examples 
// Get the main element
const mainElement = document.getElementById('mainElement');

// Example 1: Highlight parent element
function highlightParent() {
    const parent = mainElement.parentElement;
    if (parent) {
        parent.style.backgroundColor = '#fdbb2d';
        return 'Parent highlighted successfully!';
    }
    return 'No parent element found!';
}

// Example 2: Count children
function countChildren() {
    const children = mainElement.children;
    return `This element has ${children.length} children`;
}

// Example 3: Highlight siblings
function highlightSiblings() {
    const prevSibling = mainElement.previousElementSibling;
    const nextSibling = mainElement.nextElementSibling;
    
    if (prevSibling) prevSibling.style.backgroundColor = '#1a2a6c';
    if (nextSibling) nextSibling.style.backgroundColor = '#b21f1f';
    
    return 'Siblings highlighted!';
}
